CREATE TABLE popular_recipes (
recipe_id INT, 
user_id INT, 
rating DOUBLE, 
date DATE, 
name TEXT, 
PRIMARY KEY((recipe_id),date)
) WITH CLUSTERING ORDER BY (date ASC);

CREATE INDEX ON popular_recipes (date); 

CREATE TABLE recipes_by_name ( 
recipe_id INT, 
name TEXT, 
PRIMARY KEY(name,recipe_id)
) WITH CLUSTERING ORDER BY (recipe_id ASC);

CREATE TABLE recipes_by_difficulty ( 
recipe_id INT, 
rating DOUBLE, 
n_steps INT, 
minutes INT, 
category TEXT,
name TEXT,
PRIMARY KEY((category),rating,recipe_id)
) WITH CLUSTERING ORDER BY (rating DESC,recipe_id ASC);

CREATE TABLE recipes_by_tag ( 
recipe_id INT, 
rating DOUBLE, 
tags TEXT,
name TEXT, 
submitted DATE,
PRIMARY KEY((tags),rating,submitted)
) WITH CLUSTERING ORDER BY (rating DESC, submitted DESC);

CREATE TABLE recipe_details ( 
recipe_id INT, 
name TEXT, 
contributor_id INT,
submitted DATE,  
rating DOUBLE, 
tags TEXT,
n_steps INT, 
minutes INT, 
steps TEXT, 
n_ingredients INT,
ingredients TEXT, 
nutrition TEXT, 
category TEXT,
description TEXT,
PRIMARY KEY((name),recipe_id)
) WITH CLUSTERING ORDER BY (recipe_id ASC);
